    def make_gadget(depth=4):
        # base seed
        T = [(0.0, 1.0)]
        bridge_factors = [(1, 5), (12, 16), (4, 9), (8, 13)]
        # cycle through four distinct start‐factor patterns for increased asymmetry
        patterns = [
            (2, 6, 10, 14),
            (3, 7, 11, 15),
            (4, 8, 12, 16),
            (5, 9, 13, 17),
        ]
        for i in range(depth):
            lo = min(l for l, r in T)
            hi = max(r for l, r in T)
            span = hi - lo
            S = []
            starts = patterns[i % 4]
            for s in starts:
                offset = span * s - lo
                for (l, r) in T:
                    S.append((l + offset, r + offset))
            for a, b in bridge_factors:
                S.append((span * a, span * b))
            T = S
        return T
